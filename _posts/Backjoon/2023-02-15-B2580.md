### [백준] 2580 스도쿠
체감 난이도 : 🎈🎈🎈 **/5**   

DFS 로 구현하여 나타낸 방법.     

```python
def promising(i, j, N):

    for p in range(9):
        if N == arr[i][p] and p != j:
            return False
        if N == arr[p][j] and p != i:
            return False
    x, y = i // 3, j //3
    for p in range(3):
        for l in range(3):
            if N == arr[x*3+p][y*3+l] and (i, j) != (x*3+p, y*3+l):
                return False
    return True

def DFS(M):
    global arr 
    if M >= len(points):
        for _ in arr:
            print(*_)
        exit()
    else:
        ni, nj = points[M]
        for N in range(1,10):
            if promising(ni, nj, N):
                arr[ni][nj] = N
                DFS(M+1)
                arr[ni][nj] = 0


arr = [list(map(int,input().split())) for _ in range(9)]
points= []
for i in range(9):
    for j in range(9):
        if arr[i][j] == 0:
            points.append((i,j))

DFS(0)

```
