```python
def bubble_sort(arr:list):
    N = len(arr)
    sorted_list = [0]*N
    for i in range(N-1, 0, -1):
        max_temp = arr[:i+1][0]
        for j in range(1, i):            
            if max_temp < arr[:i+1][j]:
                max_temp = arr[:i+1][j]
        sorted_list[i] = max_temp
    return sorted_list

```

```python
def selection_sort(arr:list):
    N = len(arr)
```
